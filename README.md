  # Getting started

      cd client
      npm install
      npm start
      cd ..
      cd server
      npm install
      node index.js
      cd ..
      cd ML
      pip install requirements.txt
      python app.py

  server running in port => 5000
  client running in port => 3000
  ML running in port => 8000
